反转二叉树
```javascript
var invertTree = function(root) {
  // 若传入节点为null，直接返回
  if (root === null) return null

  // 从根结点开始，递归遍历左右子树
  let left = invertTree(root.left)
    right = invertTree(root.right)
    
  // 从叶子结点开始反转二叉树
  root.left = right
  root.right = left
  return root  
};
```

时间复杂度： O(n)
空间复杂度： O(n)

三数之和
```javascript
const arr = [-10, -2, 1, 2, 3, 3, 5, 6, 7, 8]
const sumOfThree = (arr) => {
  const copy = JSON.parse(JSON.stringify(arr))
  copy.sort((a,b) => a - b)
  const res = []
  for(let i = 0 ; i < copy.length; i++) {
    // unique i
    if (i > 0 && copy[i] === copy[i-1]) {
      continue
    }
    let j = i + 1, k = copy.length - 1
    while(j < k) {
      let sum = copy[i] + copy[j] + copy[k]
      if (sum === 0) {
        res.push([i, j, k])
        //j++
        // while(arr[j] === arr[j-1]) { j++ }
        while (copy[j] == copy[j+1]) j++; // j去重
        while (copy[k] == copy[k-1]) k--; // k去重
        j++;
        k--;
      } else if (sum < 0) {
        j++
      } else {
        k--
      }
    }
    
  }
return res
}
```
