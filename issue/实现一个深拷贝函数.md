> 浅拷贝：是引用的拷贝， 拷贝前后的引用数据类型存储在相同的堆区内存。
> 浅拷贝：拷贝前后的变量会存储在不同的内存区域， 前后变量的修改不会互相影响。

深拷贝实现
```javascript
// 将目标obj进行值的拷贝
const obj = { a: '1', b: [{c: ['ccc', [{d: 'test'}]]}]}

const getType = (obj) => Object.prototype.toString.call(obj).slice(8, -1)

const deepClone = (obj) => {
  let res
  if (getType(obj) === 'Object') {
    res = {}
    for (let key in obj) {
      if (['Array', 'Object'].includes(getType(obj[key]))) {
        res[key] = deepClone(obj[key])
      } else {
        res[key] = obj[key]
      }
    }
  } else if (getType(obj) === 'Array') {
    res = []
    for (let key of obj) {
      if (['Array', 'Object'].includes(getType(key))) {
        res.push(deepClone(key))
      } else {
        res.push(key)
      }
    }
  } else {
    res = obj // number、string、null、regExp、undefined、symbol
  }
  return res
}

```
