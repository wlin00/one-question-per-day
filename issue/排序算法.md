**1、冒泡排序**
（1）基础版
```typescript
  const arr = [1,3,2,4,6,5,8,7]
  const bubbleSort = (arr: any[]) => {
    for (let i = 0; i < arr.length - 1; i++) {
      for (let j = 0; j < arr.length - 1 - i; j++) {
        const tmp = arr[j]
        arr[j] = arr[j + 1]
        arr[j + 1] = tmp
      }
    }
    return arr
  }
```

（2）标记优化：若本轮次遍历没有发生排序，则说明排序完毕 直接return
```typescript
  const arr = [1,3,2,4,6,5,8,7]
  const bubbleSort = (arr: any[]) => {
    for (let i = 0; i < arr.length - 1; i++) {
      let flag = false
      for (let j = 0; j < arr.length - 1 - i; j++) {
        if (arr[j] > arr[j + 1]) {
          const tmp = arr[j]
          arr[j] = arr[j + 1]
          arr[j + 1] = tmp
          flag = true
        }
      }
      if (!flag) { // 若本轮遍历没有发生排序，则说明已经排序好了，可以break退出
        break
      }
    }
    return arr
  }
```


**2、快速排序**
（1）递归版
```typescript
  const arr = [1,3,2,4,6,5,8,7]
  const quickSort = (arr: any[]) => {
    // 递归退出条件
    if (arr.length <= 1) {
      return arr
    }
    const base = arr[0]
    const left = [], right = []
    for (let i = 1; i < arr.length; i++) {
      if (arr[i] < base) {
        left.push(arr[i])
      } else {
        right.push(arr[i])
      }
    }
    return [...quickSort(left), base, ...quickSort(right)]
  }
```
